181     /**
182     * Calculates the dot product between the calling vector and parameter v
183     * @param {Vector2} v input vector
184     * @return {number}
185     */
186     dot: function (v) {
187         return this.x * v.x + this.y * v.y;
188     },
189 
190 	/**
191 	* Returns a new vector that's perpendicular to this vector
192 	* @return {Vector2}
193 	*/
194 	perp: function () {

196 	},

198     /**

200     * @return {Vector2}
201     */
202     normalize: function () {
203         var length, inverseLength;
204         
205         length = this.length();
206         if (length < MathHelper.zeroTolerance) {
207             return new Vector2(0.0, 0.0);
208         }
209

211         return new Vector2(this.x * inverseLength,

217     * @return {number}

225     * check is required, since this is more performant than the length() method
226 	* @return {number}

741     * @return {number}
742     */
743     determinant: function () {
744         var a, b, c, d, e, f, g, h, i, j, k, l;
745         a = this.m11 * this.m22 - this.m12 * this.m21;
746         b = this.m11 * this.m23 - this.m13 * this.m21;
747         c = this.m11 * this.m24 - this.m14 * this.m21;
748         d = this.m12 * this.m23 - this.m13 * this.m22;
749         e = this.m12 * this.m24 - this.m14 * this.m22;
750         f = this.m13 * this.m24 - this.m14 * this.m23;
751         g = this.m31 * this.m42 - this.m32 * this.m41;
752         h = this.m31 * this.m43 - this.m33 * this.m41;
753         i = this.m31 * this.m44 - this.m34 * this.m41;
754         j = this.m32 * this.m43 - this.m33 * this.m42;
755         k = this.m32 * this.m44 - this.m34 * this.m42;
756         l = this.m33 * this.m44 - this.m34 * this.m43;
757         return a * l - b * k + c * j + d * i - e * h + f * g;
758     },

760     /**
761     * Returns the inverse of the calling matrix.  If the matrix cannot be inverted
762     * the the identity matrix is returned.
763     * @return {Matrix4x4}
764     */
765     inverse: function () {
766         var a, b, c, d, e, f, g, h, i, j, k, l, determinant, invD, 
767             m11, m12, m13, m14, m21, m22, m23, m24, m31, m32, m33, m34, m41, m42, m43, m44;
768             
769         a = this.m11 * this.m22 - this.m12 * this.m21;
770         b = this.m11 * this.m23 - this.m13 * this.m21;
771         c = this.m11 * this.m24 - this.m14 * this.m21;
772         d = this.m12 * this.m23 - this.m13 * this.m22;
773         e = this.m12 * this.m24 - this.m14 * this.m22;
774         f = this.m13 * this.m24 - this.m14 * this.m23;
775         g = this.m31 * this.m42 - this.m32 * this.m41;
776         h = this.m31 * this.m43 - this.m33 * this.m41;
777         i = this.m31 * this.m44 - this.m34 * this.m41;
778         j = this.m32 * this.m43 - this.m33 * this.m42;
779         k = this.m32 * this.m44 - this.m34 * this.m42;
780         l = this.m33 * this.m44 - this.m34 * this.m43;
781         determinant =  a * l - b * k + c * j + d * i - e * h + f * g;
782         if (MathHelper.abs(determinant) < MathHelper.zeroTolerance) {
783             return Matrix4x4.createIdentity();
784         }
785         
786         m11 = this.m22 * l - this.m23 * k + this.m24 * j;
787         m12 = -this.m12 * l + this.m13 * k - this.m14 * j;
788         m13 = this.m42 * f - this.m43 * e + this.m44 * d;
789         m14 = -this.m32 * f + this.m33 * e - this.m34 * d;
790         
791         m21 = -this.m21 * l + this.m23 * i - this.m24 * h;
792         m22 = this.m11 * l - this.m13 * i + this.m14 * h;
793         m23 = -this.m41 * f + this.m43 * c - this.m44 * b;
794         m24 = this.m31 * f - this.m33 * c + this.m34 * b;
795         
796         m31 = this.m21 * k - this.m22 * i + this.m24 * g;
797         m32 = -this.m11 * k + this.m12 * i - this.m14 * g;
798         m33 = this.m41 * e - this.m42 * c + this.m44 * a;
799         m34 = -this.m31 * e + this.m32 * c - this.m34 * a;
800         
801         m41 = -this.m21 * j + this.m22 * h - this.m23 * g;
802         m42 = this.m11 * j - this.m12 * h + this.m13 * g;
803         m43 = -this.m41 * d + this.m42 * b - this.m43 * a;
804         m44 = this.m31 * d - this.m32 * b + this.m33 * a;
805         invD = 1.0 / determinant;
806         return new Matrix4x4(m11 * invD, m12 * invD, m13 * invD, m14 * invD,
807                              m21 * invD, m22 * invD, m23 * invD, m24 * invD,
808                              m31 * invD, m32 * invD, m33 * invD, m34 * invD,
809                              m41 * invD, m42 * invD, m43 * invD, m44 * invD);
810     },

813     * Returns a string containing the current state of the matrix.  Useful for debugging purposes
814     * @return {string}
815     */
816     toString: function () {
817         return this.m11 + ", " + this.m12 + ", " + this.m13 + ", " + this.m14 + "\n" +
818                this.m21 + ", " + this.m22 + ", " + this.m23 + ", " + this.m24 + "\n" +
819                this.m31 + ", " + this.m32 + ", " + this.m33 + ", " + this.m34 + "\n" +
820                this.m41 + ", " + this.m42 + ", " + this.m43 + ", " + this.m44 + "\n";
821     }

846 /**
847 * Creates a Quaternion from a rotation matrix
848 * @param {Matrix4x4} m rotation matrix  
849 * @return {Quaternion} 
850 */
851 Quaternion.fromRotationMatrix = function (m) {
852     //See: http://www.euclideanspace.com/maths/geometry/rotations/conversions/matrixToQuaternion/index.htm
853 
854 	/*jslint onevar: false */
855 	//JSLint and I don't agree, as type annotions mean we need multiple vars.
856 
857 	/** @type {number} */
858     var trace; 
859 
860 	/** @type {number} */
861 	var temp; 
862 
863 	/** @type {Quaternion} */
864 	var result;
865 
866 	/** @type {number} */
867 	var largestIndex;
868     
869     
870     result = new Quaternion(0, 0, 0, 0);
871     trace = m.m11 + m.m22 + m.m33;
872     if (MathHelper.abs(trace) > MathHelper.zeroTolerance) {
873         result.w = MathHelper.sqrt(trace + 1) * 0.5;
874         temp = 1.0 / (4 * result.w);
875         result.x = (m.m32 - m.m23) * temp;
876         result.y = (m.m13 - m.m31) * temp;
877         result.z = (m.m21 - m.m12) * temp;
878     }
879     else {
880         largestIndex = 0;
881         if (m.m22 > m.m11) {
882             largestIndex = 1;
883             if (m.m33 > m.m22) {
884                 largestIndex = 2;
885             }
886         }
887         else if (m.m33 > m.m11) {
888             largestIndex = 2;
889         }
890        
891 	    switch (largestIndex) {
892         case 0:
893             result.x = 0.5 * MathHelper.sqrt(m.m11 - m.m22 - m.m33 + 1);
894             temp = 1.0 / (4 * result.x);
895             result.w = (m.m32 - m.m23) * temp;
896             result.y = (m.m12 + m.m21) * temp;
897             result.z = (m.m13 + m.m31) * temp;
898             break;
899         case 1:
900             result.y = 0.5 * MathHelper.sqrt(m.m22 - m.m11 - m.m33 + 1);
901             temp = 1.0 / (4 * result.y);
902             result.w = (m.m13 - m.m31) * temp;
903             result.x = (m.m12 + m.m21) * temp;
904             result.z = (m.m23 + m.m32) * temp;
905             break;
906         case 2:
907             result.z = 0.5 * MathHelper.sqrt(m.m33 - m.m11 - m.m22 + 1);
908             temp = 1.0 / (4 * result.z);
909             result.w = (m.m21 - m.m12) * temp;
910             result.x = (m.m13 + m.m31) * temp;
911             result.y = (m.m32 + m.m23) * temp;
912             break;
913 		}
914     }
915     return result;
916 };

931 /**
932 * Returns a quaternion that has been slerped between source and target by t amount
933 * @param {number} t A value between 0.0 and 1.0 inclusive
934 * @param {Quaternion} source The starting quaternion value
935 * @param {Quaternion} target The target quaternion value
936 * @return {Quaternion}
937 */
938 Quaternion.slerp = function (t, source, target) {
939     var cos, angle, sin, invSin, a, b;
940     
941     if (t === 0.0) {
942         return source;
943     }
944     if (t === 1.0) {
945         return target;
946     }
947     
948     cos = source.dot(target);
949     angle = MathHelper.acos(cos);
950     if (MathHelper.abs(angle) >= MathHelper.zeroTolerance) {
951         sin = MathHelper.sin(angle);
952         invSin = 1.0 / sin;
953         a = MathHelper.sin((1.0 - t) * angle) * invSin;
954         b = MathHelper.sin(t * angle) * invSin;
955         return source.multiplyScalar(a).add(target.multiplyScalar(b));
956     }
957     
958     return source;
959 };
