120 Vector2.prototype =
121 {
122     /**
123     * Calculates the dot product between the calling vector and parameter v
124     * @param {Vector2} v input vector
125     * @return {Number}
126     */
127     dot: function (v) {
128         return this.x * v.x + this.y * v.y;
129     },
130 
131 	/**
132 	* Returns a new vector that's perpendicular to this vector
133 	* @return {Vector2}
134 	*/
135 	perp: function () {
136 		return new Vector2(this.y, -this.x);
137 	},
138 
139     /**
140     * Creates a unit length version of the vector, which still points in the same direction as the original vector
141     * @return {Vector2}
142     */
143     normalize: function () {
144         var length, inverseLength;
145         
146         length = this.length();
147         if (length < MathHelper.zeroTolerance) {
148             return new Vector2(0.0, 0.0);
149         }
150 
151         inverseLength = 1.0 / length;
152         return new Vector2(this.x * inverseLength,
153                            this.y * inverseLength);
154     },
155 
156     /**
157     * Calculates the length of the vector
158     * @return {Number}
159     */
160     length: function () {
161         return MathHelper.sqrt(this.x * this.x + this.y * this.y);
162     },
163 
164     /**
165     * Calculates the length of the vector squared.  Useful if only a relative length
166     * check is required, since this is more performant than the length() method
167 	* @return {Number}
168     */
169     lengthSquared: function () {
170         return this.x * this.x + this.y * this.y;
171     },
172 
173     /**
174     * Adds vector v to the current vector and returns the result.
175     * @param {Vector2} v input vector
176     * @returns {Vector2} A vector containing the addition of the two input vectors
177     */
178     add: function (v) {
179         return new Vector2(this.x + v.x,
180                            this.y + v.y);
181     },
182 
183     /**
184     * Subtracts vector v from the current vector and returns the result.
185     * @param {Vector2} v input vector
186     * @returns {Vector2} A vector containing the subtraction of the two input vectors
187     */
188     subtract: function (v) {
189         return new Vector2(this.x - v.x,
190                            this.y - v.y);
191     },
192 
193     /**
194     * Multiplies each element of the vector with scalar f and returns the result
195     * @param {Number} f a value that will be multiplied with each element of the vector
196     * @return {Vector2}
197     */
198     multiplyScalar: function (f) {
199         return new Vector2(this.x * f,
200                            this.y * f);
201     },
202 
203     /**
204     * Checks if the calling vector is equal to parameter vector v
205     * @param {Vector2} v input vector
206     * @returns {Boolean} A Boolean value, true if each element of the calling vector match input vector v, false otherwise
207     */
208     equals: function (v) {
209         return this.x === v.x &&
210                this.y === v.y;
211     },

213     /**
214     * Returns a string containing the current state of the vector, useful for debugging purposes
215     * @return {String}
216     */
217     toString: function () {
218         return '[' + this.x + ', ' + this.y + ']';
219     }

464 /**

484 function Matrix4x4(m11, m12, m13, m14,
485                    m21, m22, m23, m24,

487                    m41, m42, m43, m44) {

545 /**
546 * Returns a matrix that rotates a vector around the x axis, from the origin. The rotation matrix

549 * @return {Matrix4x4}

552     return new Matrix4x4(1, 0, 0, 0,
553                          0, MathHelper.cos(angle), -MathHelper.sin(angle), 0,
554                          0, MathHelper.sin(angle), MathHelper.cos(angle), 0,
555                          0, 0, 0, 1);

672     /**
673     * Multiples the matrix by the column vector v. It is assumed the Vector3 v value
674     * is equivalent to a Vector4 instance with a w value of 0
675     * @param {Vector3} v input vector
676     * @return {Vector3}
677     */
678     transformVector3: function (v) {
679         return new Vector3(this.m11 * v.x + this.m12 * v.y + this.m13 * v.z,
680                            this.m21 * v.x + this.m22 * v.y + this.m23 * v.z,
681                            this.m31 * v.x + this.m32 * v.y + this.m33 * v.z);
682     },
683     
684     /**
685     * Returns the determinant of the calling matrix
686     * @return {Number}
687     */
688     determinant: function () {
689         var a, b, c, d, e, f, g, h, i, j, k, l;
690         a = this.m11 * this.m22 - this.m12 * this.m21;
691         b = this.m11 * this.m23 - this.m13 * this.m21;
692         c = this.m11 * this.m24 - this.m14 * this.m21;
693         d = this.m12 * this.m23 - this.m13 * this.m22;
694         e = this.m12 * this.m24 - this.m14 * this.m22;
695         f = this.m13 * this.m24 - this.m14 * this.m23;
696         g = this.m31 * this.m42 - this.m32 * this.m41;
697         h = this.m31 * this.m43 - this.m33 * this.m41;
698         i = this.m31 * this.m44 - this.m34 * this.m41;
699         j = this.m32 * this.m43 - this.m33 * this.m42;
700         k = this.m32 * this.m44 - this.m34 * this.m42;
701         l = this.m33 * this.m44 - this.m34 * this.m43;
702         return a * l - b * k + c * j + d * i - e * h + f * g;
703     },
704     
705     /**
706     * Returns the inverse of the calling matrix.  If the matrix cannot be inverted
707     * the the identity matrix is returned.
708     * @return {Matrix4x4}
709     */
710     inverse: function () {
711         var a, b, c, d, e, f, g, h, i, j, k, l, determinant, invD, 
712             m11, m12, m13, m14, m21, m22, m23, m24, m31, m32, m33, m34, m41, m42, m43, m44;
713             
714         a = this.m11 * this.m22 - this.m12 * this.m21;
715         b = this.m11 * this.m23 - this.m13 * this.m21;
716         c = this.m11 * this.m24 - this.m14 * this.m21;
717         d = this.m12 * this.m23 - this.m13 * this.m22;
718         e = this.m12 * this.m24 - this.m14 * this.m22;
719         f = this.m13 * this.m24 - this.m14 * this.m23;
720         g = this.m31 * this.m42 - this.m32 * this.m41;
721         h = this.m31 * this.m43 - this.m33 * this.m41;
722         i = this.m31 * this.m44 - this.m34 * this.m41;
723         j = this.m32 * this.m43 - this.m33 * this.m42;
724         k = this.m32 * this.m44 - this.m34 * this.m42;
725         l = this.m33 * this.m44 - this.m34 * this.m43;
726         determinant =  a * l - b * k + c * j + d * i - e * h + f * g;
727         if (MathHelper.abs(determinant) < MathHelper.zeroTolerance)
728         {
729             return Matrix4x4.createIdentity();
730         }
731         
732         m11 = this.m22 * l - this.m23 * k + this.m24 * j;
733         m12 = -this.m12 * l + this.m13 * k - this.m14 * j;
734         m13 = this.m42 * f - this.m43 * e + this.m44 * d;
735         m14 = -this.m32 * f + this.m33 * e - this.m34 * d;
736         
737         m21 = -this.m21 * l + this.m23 * i - this.m24 * h;
738         m22 = this.m11 * l - this.m13 * i + this.m14 * h;
739         m23 = -this.m41 * f + this.m43 * c - this.m44 * b;
740         m24 = this.m31 * f - this.m33 * c + this.m34 * b;
741         
742         m31 = this.m21 * k - this.m22 * i + this.m24 * g;
743         m32 = -this.m11 * k + this.m12 * i - this.m14 * g;
744         m33 = this.m41 * e - this.m42 * c + this.m44 * a;
745         m34 = -this.m31 * e + this.m32 * c - this.m34 * a;
746         
747         m41 = -this.m21 * j + this.m22 * h - this.m23 * g;
748         m42 = this.m11 * j - this.m12 * h + this.m13 * g;
749         m43 = -this.m41 * d + this.m42 * b - this.m43 * a;
750         m44 = this.m31 * d - this.m32 * b + this.m33 * a;
751         invD = 1.0 / determinant;
752         return new Matrix4x4(m11 * invD, m12 * invD, m13 * invD, m14 * invD,
753                              m21 * invD, m22 * invD, m23 * invD, m24 * invD,
754                              m31 * invD, m32 * invD, m33 * invD, m34 * invD,
755                              m41 * invD, m42 * invD, m43 * invD, m44 * invD);
756     },
757 
758     /**
759     * Returns a string containing the current state of the matrix.  Useful for debugging purposes
760     * @return {String}
761     */
762     toString: function () {
763         return this.m11 + ", " + this.m12 + ", " + this.m13 + ", " + this.m14 + "\n" +
764                this.m21 + ", " + this.m22 + ", " + this.m23 + ", " + this.m24 + "\n" +
765                this.m31 + ", " + this.m32 + ", " + this.m33 + ", " + this.m34 + "\n" +
766                this.m41 + ", " + this.m42 + ", " + this.m43 + ", " + this.m44 + "\n";
767     }
768 };

788 Quaternion.prototype = 
789 {
790     dot: function(q)
791     {
792         return this.w * q.w + this.x * q.x + this.y * q.y + this.z * q.z;
793     },
794     
795     inverse: function() {
796     
797         var norm = this.w * this.w + this.x * this.x + this.y * this.y * this.z * this.z;
798         if(norm > 0.0)
799         {
800             var inverseNorm = 1.0 / norm;
801             return new Quaternion(this.w * inverseNorm,
802                                   this.x * inverseNorm,
803                                   this.y * inverseNorm,
804                                   this.z * inverseNorm)
805         }
806         return new Quaternion(0.0, 0.0, 0.0, 0.0);
807     },
808     
809     conjugate: function() {
810         return new Quaternion(this.w, -this.x, -this.y, -this.z);
811     },
812     
813     slerp: function(t, source, target) {
814     
815         var cos = source.dot(target);
816         var angle = MathHelper.acos(cos);
817         
818         if(MathHelper.fabs(angle) >= MathHelper.zeroTolerance)
819         {
820             var sin = MathHelper.sin(angle);
821             var invSin = 1.0 / sin;
822             var a = MathHelper.sin((1.0 - t) * angle) * invSin;
823             var b = MathHelper.sin(t * angle) * invSin;
824             return source.multiplyScalar(a).add(target.multiplyScalar(b));
825         }
826         
827         return source;
828     },
829     
830     //Rotates a Vector3 by the quaternion
831     rotateVector: function(v) {
832         var m = toRotationMatrix();
833         return m.multiplyVector3(v);
834     },
835     
836     add: function(q) {
837         return new Quaternion(this.w + q.w, this.x + q.x, this.y + q.y, this.z + q.z);
838     },
839     
840     multiplyScalar: function(f, q) {
841         return new Quaternion(this.w * f, this.x * f, this.y * f, this.z * f);
842     },
843     
844     toRotationMatrix: function() {
845     
846         var x = 2.0 * this.x;
847         var y = 2.0 * this.y;
848         var z = 2.0 * this.z;
849         var wx = x * this.w;
850         var wy = y * this.w;
851         var wz = z * this.w;
852         var xx = x * this.x;
853         var xy = y * this.x;
854         var xz = z * this.x;
855         var yy = y * this.y;
856         var yz = z * this.y;
857         var zz = z * this.z;
858         
859         return new Matrix4x4(1.0 - (yy + zz), xy - wz, xz + wy, 0,
860                              xy + wz, 1.0 - (xx + zz), yz - wx, 0,
861                              xz - wy, yz + wx, 1.0 - (xx + yy), 0,
862                              0, 0, 0, 1);
863     },
864     
865     // Returns the result as a Vector4, the w component is the angle
866     toAxisAngle: function() {
867     
868         var lengthSquared = this.x * this.x + this.y * this.y + this.z * this.z;
869         if(lengthSquared > MathHelper.zeroTolerance)
870         {
871             var inverseLength = MathHelper.invSqrt(lengthSquared);
872             return new Vector4(this.x * inverseLength, this.y * inverseLength, this.z * inverseLength, 2.0 * MathHelper.acos(this.w));
873         }
874         return new Vector4(1, 0, 0, 0);
875     }
