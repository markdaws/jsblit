 83 * A vector class representing three dimensional space
 84 * @constructor
 85 * @param {Number} x
 86 * @param {Number} y
 87 * @param {Number} z
 88 */
 89 function Vector3(x, y, z)
 90 {
 91     this.x = x;
 92     this.y = y;
 93     this.z = z;
 94 }
 95 
 96 Vector3.prototype =
 97 {
 98     /**
 99     * Calculates the dot product between the calling vector and parameter v
100     * @param {Vector3} v input vector
101     * @return {Number}
102     */
103     dot: function (v) {
104         return this.x * v.x + this.y * v.y + this.z * v.z;
105     },
106 
107     /**
108     * Creates a unit length version of the vector, which still points in the same direction as the original vector
109     * @return {Vector3}
110     */
111     normalize: function () {
112         var length, inverseLength;
113         
114         length = this.length();
115         if (length < MathHelper.zeroTolerance) {
116             return new Vector3(0.0, 0.0, 0.0);
117         }
118 
119         inverseLength = 1.0 / length;
120         return new Vector3(this.x * inverseLength,
121                            this.y * inverseLength,
122                            this.z * inverseLength);
123     },
124 
125     /**
126     * Calculates the cross product of the vector and vector parameter v and returns the result
127     * @param {Vector3} v input vector
128     * @return {Vector3}
129     */
130     cross: function (v) {
131         return new Vector3(this.y * v.z - this.z * v.y,
132                            this.z * v.x - this.x * v.z,
133                            this.x * v.y - this.y * v.x);
134     },
135 
136     /**
137     * Calculates the length of the vector
138     * @return {Number}
139     */
140     length: function () {
141         return MathHelper.sqrt(this.x * this.x + this.y * this.y + this.z * this.z);
142     },
143 
144     /**
145     * Calculates the length of the vector squared.  Useful if only a relative length
146     * check is required, since this is more performant than the length() method
147     * @return {Number}
148     */
149     lengthSquared: function () {
150         return this.x * this.x + this.y * this.y + this.z * this.z;
151     },
152 
153     /**
154     * Adds vector v to the current vector and returns the result.
155     * @param {Vector3} v input vector
156     * @returns {Vector3} A vector containing the addition of the two input vectors
157     */
158     add: function (v) {
159         return new Vector3(this.x + v.x,
160                            this.y + v.y,
161                            this.z + v.z);
162     },
163 
164     /**
165     * Subtracts vector v from the current vector and returns the result.
166     * @param {Vector3} v input vector
167     * @returns {Vector3} A vector containing the subtraction of the two input vectors
168     */
169     subtract: function (v) {
170         return new Vector3(this.x - v.x,
171                            this.y - v.y,
172                            this.z - v.z);
173     },
174 
175     /**
176     * Multiplies each element of the vector with scalar f and returns the result
177     * @param {Number} f a value that will be multiplied with each element of the vector
178     * @return {Vector3}
179     */
180     multiplyScalar: function (f) {
181         return new Vector3(this.x * f,
182                            this.y * f,
183                            this.z * f);
184     },
185 
186     /**
187     * Checks if the calling vector is equal to parameter vector v
188     * @param {Vector3} v input vector
189     * @return {Boolean} A Boolean value, true if each element of the calling vector match input vector v, false otherwise
190     */
191     equals: function (v) {
192         return this.x === v.x &&
193                this.y === v.y &&
194                this.z === v.z;
195     },
196     
197     /**
198     * Returns a string containing the current state of the vector, useful for debugging purposes
199     * @return {String}
200     */
201     toString: function () {
202         return '[' + this.x + ', ' + this.y + ', ' + this.z + ']';
203     }
204 };

325 /**
326 * Creates a 4x4 matrix
327 * @constructor
328 * @param {Number} m11
329 * @param {Number} m12
330 * @param {Number} m13
331 * @param {Number} m14
332 * @param {Number} m21
333 * @param {Number} m22
334 * @param {Number} m23
335 * @param {Number} m24
336 * @param {Number} m31
337 * @param {Number} m32
338 * @param {Number} m33
339 * @param {Number} m34
340 * @param {Number} m41
341 * @param {Number} m42
342 * @param {Number} m43
343 * @param {Number} m44
344 */
345 function Matrix4x4(m11, m12, m13, m14,
346                    m21, m22, m23, m24,
347                    m31, m32, m33, m34,
348                    m41, m42, m43, m44) {
349     this.m11 = m11;
350     this.m12 = m12;
351     this.m13 = m13;
352     this.m14 = m14;
353     this.m21 = m21;
354     this.m22 = m22;
355     this.m23 = m23;
356     this.m24 = m24;
357     this.m31 = m31;
358     this.m32 = m32;
359     this.m33 = m33;
360     this.m34 = m34;
361     this.m41 = m41;
362     this.m42 = m42;
363     this.m43 = m43;
364     this.m44 = m44;
365 }

367 /**
368 * Returns an identity matrix
369 * @return {Matrix4x4}
370 */
371 Matrix4x4.createIdentity = function () {
372     return new Matrix4x4(1, 0, 0, 0,
373                          0, 1, 0, 0,
374                          0, 0, 1, 0,
375                          0, 0, 0, 1);
376 };

378 /**
379 * Returns a scaling matrix
380 * @param {Number} sx The x scaling factor
381 * @param {Number} sy The y scaling factor
382 * @param {Number} sz The z scaling factor
383 * @return {Matrix4x4}
384 */
385 Matrix4x4.createScale = function (sx, sy, sz) {
386     return new Matrix4x4(sx, 0,  0,  0,
387                          0,  sy, 0,  0,
388                          0,  0,  sz, 0,
389                          0,  0,  0,  1);
390 };

392 /**
393 * Returns a translation matrix to be used with a column vector p = M * v
394 * @param {Number} tx The x translation value
395 * @param {Number} ty The y translation value
396 * @param {Number} tz The z translation value
397 * @return {Matrix4x4}
398 */
399 Matrix4x4.createTranslate = function (tx, ty, tz) {
400     return new Matrix4x4(1, 0, 0, tx,
401                          0, 1, 0, ty,
402                          0, 0, 1, tz,
403                          0, 0, 0, 1);

406 /**
407 * Returns a matrix that rotates a vector around the x axis, from the origin. The rotation matrix
408 * is a right handed rotation, a positive angle will rotate the vector anticlockwise around the x axis
409 * @param {Number} angle The angle to rotate in radians
410 * @return {Matrix4x4}
411 */
412 Matrix4x4.rotateX = function (angle) {
413     return new Matrix4x4(1, 0, 0, 0,
414                          0, MathHelper.cos(angle), -MathHelper.sin(angle), 0,
415                          0, MathHelper.sin(angle), MathHelper.cos(angle), 0,
416                          0, 0, 0, 1);
417 };
418 
419 /**
420 * Returns a matrix that rotates a vector around the y axis, from the origin. The rotation matrix
421 * is a right handed rotation, a positive angle will rotate the vector anticlockwise around the y axis
422 * @param {Number} angle The angle to rotate in radians
423 * @return {Matrix4x4}
424 */
425 Matrix4x4.rotateY = function (angle) {
426     return new Matrix4x4(MathHelper.cos(angle), 0, -MathHelper.sin(angle), 0,
427                          0, 1, 0, 0,
428                          MathHelper.sin(angle), 0, MathHelper.cos(angle), 0,
429                          0, 0, 0, 1);
430 };
431 
432 /**
433 * Returns a matrix that rotates a vector around the z axis, from the origin. The rotation matrix
434 * is a right handed rotation, a positive angle will rotate the vector anticlockwise around the z axis
435 * @param {Number} angle The angle to rotate in radians
436 * @return {Matrix4x4}
437 */
438 Matrix4x4.rotateZ = function (angle) {
439     return new Matrix4x4(MathHelper.cos(angle), -MathHelper.sin(angle), 0, 0,
440                          MathHelper.sin(angle), MathHelper.cos(angle), 0, 0,
441                          0, 0, 1, 0,
442                          0, 0, 0, 1);
443 };

445 Matrix4x4.prototype =
446 {
447     /**
448     * Adds matrix m to to the current matrix and returns the result
449     * @param {Matrix4x4} m The matrix which will be added to the calling matrix
450     * @return {Matrix4x4}
451     */
452     add: function (m) {
453         return new Matrix4x4(this.m11 + m.m11, this.m12 + m.m12, this.m13 + m.m13, this.m14 + m.m14,
454                              this.m21 + m.m21, this.m22 + m.m22, this.m23 + m.m23, this.m24 + m.m24,
455                              this.m31 + m.m31, this.m32 + m.m32, this.m33 + m.m33, this.m34 + m.m34,
456                              this.m41 + m.m41, this.m42 + m.m42, this.m43 + m.m43, this.m44 + m.m44);
457     },
458 
459     /**
460     * Adds matrix m to to the current matrix and returns the result
461     * @param {Matrix4x4} m The matrix which will be added to the calling matrix
462     * @return {Matrix4x4}
463     */
464     subtract: function (m) {
465         return new Matrix4x4(this.m11 - m.m11, this.m12 - m.m12, this.m13 - m.m13, this.m14 - m.m14,
466                              this.m21 - m.m21, this.m22 - m.m22, this.m23 - m.m23, this.m24 - m.m24,
467                              this.m31 - m.m31, this.m32 - m.m32, this.m33 - m.m33, this.m34 - m.m34,
468                              this.m41 - m.m41, this.m42 - m.m42, this.m43 - m.m43, this.m44 - m.m44);
469     },
470 
471     /**
472     * Multiples the calling matrix by matrix m and returns the result
473     * @param {Matrix4x4} m input matrix
474     * @return {Matrix4x4}
475     */
476     multiply: function (m) {
477         return new Matrix4x4(this.m11 * m.m11 + this.m12 * m.m21 + this.m13 * m.m31 + this.m14 * m.m41,
478                              this.m11 * m.m12 + this.m12 * m.m22 + this.m13 * m.m32 + this.m14 * m.m42,
479                              this.m11 * m.m13 + this.m12 * m.m23 + this.m13 * m.m33 + this.m14 * m.m43,
480                              this.m11 * m.m14 + this.m12 * m.m24 + this.m13 * m.m34 + this.m14 * m.m44,
481 
482                              this.m21 * m.m11 + this.m22 * m.m21 + this.m23 * m.m31 + this.m24 * m.m41,
483                              this.m21 * m.m12 + this.m22 * m.m22 + this.m23 * m.m32 + this.m24 * m.m42,
484                              this.m21 * m.m13 + this.m22 * m.m23 + this.m23 * m.m33 + this.m24 * m.m43,
485                              this.m21 * m.m14 + this.m22 * m.m24 + this.m23 * m.m34 + this.m24 * m.m44,
486 
487                              this.m31 * m.m11 + this.m32 * m.m21 + this.m33 * m.m31 + this.m34 * m.m41,
488                              this.m31 * m.m12 + this.m32 * m.m22 + this.m33 * m.m32 + this.m34 * m.m42,
489                              this.m31 * m.m13 + this.m32 * m.m23 + this.m33 * m.m33 + this.m34 * m.m43,
490                              this.m31 * m.m14 + this.m32 * m.m24 + this.m33 * m.m34 + this.m34 * m.m44,
491 
492                              this.m41 * m.m11 + this.m42 * m.m21 + this.m43 * m.m31 + this.m44 * m.m41,
493                              this.m41 * m.m12 + this.m42 * m.m22 + this.m43 * m.m32 + this.m44 * m.m42,
494                              this.m41 * m.m13 + this.m42 * m.m23 + this.m43 * m.m33 + this.m44 * m.m43,
495                              this.m41 * m.m14 + this.m42 * m.m24 + this.m43 * m.m34 + this.m44 * m.m44);
496     },
497 
498     /**
499     * Multiples each element of the matrix by the scalar f and returns the result
500     * @param {Number} f input scalar
501     * @return {Matrix4x4}
502     */
503     multiplyScalar: function (f) {
504         return new Matrix4x4(this.m11 * f, this.m12 * f, this.m13 * f, this.m14 * f,
505                              this.m21 * f, this.m22 * f, this.m23 * f, this.m24 * f,
506                              this.m31 * f, this.m32 * f, this.m33 * f, this.m34 * f,
507                              this.m41 * f, this.m42 * f, this.m43 * f, this.m44 * f);
508     },
509 
510     /**
511     * Returns the transpose of the calling matrix
512     * @return {Matrix4x4}
513     */
514     transpose: function () {
515         return new Matrix4x4(this.m11, this.m21, this.m31, this.m41,
516                              this.m12, this.m22, this.m32, this.m42,
517                              this.m13, this.m23, this.m33, this.m43,
518                              this.m14, this.m24, this.m34, this.m44);
519     },
520     
521     /**
522     * Multiples the matrix by the column vector v
523     * @param {Vector4} v input vector
524     * @return {Vector4}
525     */
526     transformVector4: function (v) {
527         return new Vector4(this.m11 * v.x + this.m12 * v.y + this.m13 * v.z + this.m14 * v.w,
528                            this.m21 * v.x + this.m22 * v.y + this.m23 * v.z + this.m24 * v.w,
529                            this.m31 * v.x + this.m32 * v.y + this.m33 * v.z + this.m34 * v.w,
530                            this.m41 * v.x + this.m42 * v.y + this.m43 * v.z + this.m44 * v.w);
531     },
532     
533     /**
534     * Multiples the matrix by the column vector v. It is assumed the Vector3 v value
535     * is equivalent to a Vector4 instance with a w value of 0
536     * @param {Vector3} v input vector
537     * @return {Vector3}
538     */
539     transformVector3: function (v) {
540         return new Vector3(this.m11 * v.x + this.m12 * v.y + this.m13 * v.z,
541                            this.m21 * v.x + this.m22 * v.y + this.m23 * v.z,
542                            this.m31 * v.x + this.m32 * v.y + this.m33 * v.z);
543     },
544     
545     /**
546     * Returns the determinant of the calling matrix
547     * @return {Number}
548     */
549     determinant: function () {
550         var a, b, c, d, e, f, g, h, i, j, k, l;
551         a = this.m11 * this.m22 - this.m12 * this.m21;
552         b = this.m11 * this.m23 - this.m13 * this.m21;
553         c = this.m11 * this.m24 - this.m14 * this.m21;
554         d = this.m12 * this.m23 - this.m13 * this.m22;
555         e = this.m12 * this.m24 - this.m14 * this.m22;
556         f = this.m13 * this.m24 - this.m14 * this.m23;
557         g = this.m31 * this.m42 - this.m32 * this.m41;
558         h = this.m31 * this.m43 - this.m33 * this.m41;
559         i = this.m31 * this.m44 - this.m34 * this.m41;
560         j = this.m32 * this.m43 - this.m33 * this.m42;
561         k = this.m32 * this.m44 - this.m34 * this.m42;
562         l = this.m33 * this.m44 - this.m34 * this.m43;
563         return a * l - b * k + c * j + d * i - e * h + f * g;
564     },
565 
566     /**
567     * Returns a string containing the current state of the matrix.  Useful for debugging purposes
568     * @return {String}
569     */
570     toString: function () {
571         return this.m11 + ", " + this.m12 + ", " + this.m13 + ", " + this.m14 + "\n" +
572                this.m21 + ", " + this.m22 + ", " + this.m23 + ", " + this.m24 + "\n" +
573                this.m31 + ", " + this.m32 + ", " + this.m33 + ", " + this.m34 + "\n" +
574                this.m41 + ", " + this.m42 + ", " + this.m43 + ", " + this.m44 + "\n";
575     }
576 };
