131 /**
132 * A vector class representing two dimensional space
133 * @constructor
134 * @param {number} x
135 * @param {number} y
136 */
137 function Vector2(x, y)
138 {
139     this.x = x;
140     this.y = y;
141 }
142 
143 Vector2.prototype =
144 {
145     /**
146     * Calculates the dot product between the calling vector and parameter v
147     * @param {Vector2} v input vector
148     * @return {number}
149     */
150     dot: function (v) {
151         return this.x * v.x + this.y * v.y;
152     },
153 
154 	/**
155 	* Returns a new vector that's perpendicular to this vector
156 	* @return {Vector2}
157 	*/
158 	perp: function () {
159 		return new Vector2(this.y, -this.x);
160 	},
161 
162     /**
163     * Creates a unit length version of the vector, which still points in the same direction as the original vector
164     * @return {Vector2}
165     */
166     normalize: function () {
167         var length, inverseLength;
168         
169         length = this.length();
170         if (length < MathHelper.zeroTolerance) {
171             return new Vector2(0.0, 0.0);
172         }
173 
174         inverseLength = 1.0 / length;
175         return new Vector2(this.x * inverseLength,
176                            this.y * inverseLength);
177     },
178 
179     /**
180     * Calculates the length of the vector
181     * @return {number}
182     */
183     length: function () {
184         return MathHelper.sqrt(this.x * this.x + this.y * this.y);
185     },
186 
187     /**
188     * Calculates the length of the vector squared.  Useful if only a relative length
189     * check is required, since this is more performant than the length() method
190 	* @return {number}
191     */
192     lengthSquared: function () {
193         return this.x * this.x + this.y * this.y;
194     },
195 
196     /**
197     * Adds vector v to the current vector and returns the result.
198     * @param {Vector2} v input vector
199     * @returns {Vector2} A vector containing the addition of the two input vectors
200     */
201     add: function (v) {
202         return new Vector2(this.x + v.x,
203                            this.y + v.y);
204     },
205 
206     /**
207     * Subtracts vector v from the current vector and returns the result.
208     * @param {Vector2} v input vector
209     * @returns {Vector2} A vector containing the subtraction of the two input vectors
210     */
211     subtract: function (v) {
212         return new Vector2(this.x - v.x,
213                            this.y - v.y);
214     },
215 
216     /**
217     * Multiplies each element of the vector with scalar f and returns the result
218     * @param {number} f a value that will be multiplied with each element of the vector
219     * @return {Vector2}
220     */
221     multiplyScalar: function (f) {
222         return new Vector2(this.x * f,
223                            this.y * f);
224     },
225 
226     /**
227     * Checks if the calling vector is equal to parameter vector v
228     * @param {Vector2} v input vector
229     * @returns {boolean} A Boolean value, true if each element of the calling vector match input vector v, false otherwise
230     */
231     equals: function (v) {
232         return this.x === v.x &&
233                this.y === v.y;
234     },
235     
236     /**
237     * Returns a string containing the current state of the vector, useful for debugging purposes
238     * @return {string}
239     */
240     toString: function () {
241         return '[' + this.x + ', ' + this.y + ']';
242     }
243 };

487 /**
488 * Creates a 4x4 matrix
489 * @constructor
490 * @param {number} m11
491 * @param {number} m12
492 * @param {number} m13
493 * @param {number} m14
494 * @param {number} m21
495 * @param {number} m22
496 * @param {number} m23
497 * @param {number} m24
498 * @param {number} m31
499 * @param {number} m32
500 * @param {number} m33
501 * @param {number} m34
502 * @param {number} m41
503 * @param {number} m42
504 * @param {number} m43
505 * @param {number} m44
506 */
507 function Matrix4x4(m11, m12, m13, m14,
508                    m21, m22, m23, m24,
509                    m31, m32, m33, m34,
510                    m41, m42, m43, m44) {
511     this.m11 = m11;
512     this.m12 = m12;
513     this.m13 = m13;
514     this.m14 = m14;
515     this.m21 = m21;
516     this.m22 = m22;
517     this.m23 = m23;
518     this.m24 = m24;
519     this.m31 = m31;
520     this.m32 = m32;
521     this.m33 = m33;
522     this.m34 = m34;
523     this.m41 = m41;
524     this.m42 = m42;
525     this.m43 = m43;
526     this.m44 = m44;
527 }

529 /**
530 * Returns an identity matrix
531 * @return {Matrix4x4}
532 */
533 Matrix4x4.createIdentity = function () {
534     return new Matrix4x4(1, 0, 0, 0,
535                          0, 1, 0, 0,
536                          0, 0, 1, 0,
537                          0, 0, 0, 1);

540 /**
541 * Returns a scaling matrix

543 * @param {number} sy The y scaling factor

547 Matrix4x4.createScale = function (sx, sy, sz) {

561 Matrix4x4.createTranslation = function (tx, ty, tz) {

568 /**
569 * Returns a matrix that rotates a vector around the x axis, from the origin. The rotation matrix
570 * is a right handed rotation, a positive angle will rotate the vector anticlockwise around the x axis
571 * @param {number} angle The angle to rotate in radians
572 * @return {Matrix4x4}
573 */
574 Matrix4x4.createRotationX = function (angle) {
575     return new Matrix4x4(1, 0, 0, 0,
576                          0, MathHelper.cos(angle), -MathHelper.sin(angle), 0,
577                          0, MathHelper.sin(angle), MathHelper.cos(angle), 0,
578                          0, 0, 0, 1);
579 };

585 * @return {Matrix4x4}

587 Matrix4x4.createRotationY = function (angle) {

607 Matrix4x4.prototype =

609     /**
610     * Adds matrix m to to the current matrix and returns the result
611     * @param {Matrix4x4} m The matrix which will be added to the calling matrix
612     * @return {Matrix4x4}

614     add: function (m) {
615         return new Matrix4x4(this.m11 + m.m11, this.m12 + m.m12, this.m13 + m.m13, this.m14 + m.m14,
616                              this.m21 + m.m21, this.m22 + m.m22, this.m23 + m.m23, this.m24 + m.m24,
617                              this.m31 + m.m31, this.m32 + m.m32, this.m33 + m.m33, this.m34 + m.m34,

623     * @param {Matrix4x4} m The matrix which will be added to the calling matrix
624     * @return {Matrix4x4}
625     */
626     subtract: function (m) {
627         return new Matrix4x4(this.m11 - m.m11, this.m12 - m.m12, this.m13 - m.m13, this.m14 - m.m14,
628                              this.m21 - m.m21, this.m22 - m.m22, this.m23 - m.m23, this.m24 - m.m24,
629                              this.m31 - m.m31, this.m32 - m.m32, this.m33 - m.m33, this.m34 - m.m34,
630                              this.m41 - m.m41, this.m42 - m.m42, this.m43 - m.m43, this.m44 - m.m44);
631     },

633     /**
634     * Multiples the calling matrix by matrix m and returns the result
635     * @param {Matrix4x4} m input matrix
636     * @return {Matrix4x4}
637     */
638     multiply: function (m) {
639         return new Matrix4x4(this.m11 * m.m11 + this.m12 * m.m21 + this.m13 * m.m31 + this.m14 * m.m41,
640                              this.m11 * m.m12 + this.m12 * m.m22 + this.m13 * m.m32 + this.m14 * m.m42,
641                              this.m11 * m.m13 + this.m12 * m.m23 + this.m13 * m.m33 + this.m14 * m.m43,
642                              this.m11 * m.m14 + this.m12 * m.m24 + this.m13 * m.m34 + this.m14 * m.m44,
643 
644                              this.m21 * m.m11 + this.m22 * m.m21 + this.m23 * m.m31 + this.m24 * m.m41,
645                              this.m21 * m.m12 + this.m22 * m.m22 + this.m23 * m.m32 + this.m24 * m.m42,
646                              this.m21 * m.m13 + this.m22 * m.m23 + this.m23 * m.m33 + this.m24 * m.m43,
647                              this.m21 * m.m14 + this.m22 * m.m24 + this.m23 * m.m34 + this.m24 * m.m44,
648 
649                              this.m31 * m.m11 + this.m32 * m.m21 + this.m33 * m.m31 + this.m34 * m.m41,
650                              this.m31 * m.m12 + this.m32 * m.m22 + this.m33 * m.m32 + this.m34 * m.m42,
651                              this.m31 * m.m13 + this.m32 * m.m23 + this.m33 * m.m33 + this.m34 * m.m43,
652                              this.m31 * m.m14 + this.m32 * m.m24 + this.m33 * m.m34 + this.m34 * m.m44,
653 
654                              this.m41 * m.m11 + this.m42 * m.m21 + this.m43 * m.m31 + this.m44 * m.m41,
655                              this.m41 * m.m12 + this.m42 * m.m22 + this.m43 * m.m32 + this.m44 * m.m42,
656                              this.m41 * m.m13 + this.m42 * m.m23 + this.m43 * m.m33 + this.m44 * m.m43,
657                              this.m41 * m.m14 + this.m42 * m.m24 + this.m43 * m.m34 + this.m44 * m.m44);
658     },
659 
660     /**
661     * Multiples each element of the matrix by the scalar f and returns the result
662     * @param {number} f input scalar
663     * @return {Matrix4x4}
664     */
665     multiplyScalar: function (f) {
666         return new Matrix4x4(this.m11 * f, this.m12 * f, this.m13 * f, this.m14 * f,
667                              this.m21 * f, this.m22 * f, this.m23 * f, this.m24 * f,
668                              this.m31 * f, this.m32 * f, this.m33 * f, this.m34 * f,
669                              this.m41 * f, this.m42 * f, this.m43 * f, this.m44 * f);
670     },
671 
672     /**
673     * Returns the transpose of the calling matrix
674     * @return {Matrix4x4}
675     */
676     transpose: function () {
677         return new Matrix4x4(this.m11, this.m21, this.m31, this.m41,
678                              this.m12, this.m22, this.m32, this.m42,
679                              this.m13, this.m23, this.m33, this.m43,
680                              this.m14, this.m24, this.m34, this.m44);
681     },
682     
683     /**
684     * Multiples the matrix by the column vector v
685     * @param {Vector4} v input vector
686     * @return {Vector4}
687     */
688     transformVector4: function (v) {
689         return new Vector4(this.m11 * v.x + this.m12 * v.y + this.m13 * v.z + this.m14 * v.w,
690                            this.m21 * v.x + this.m22 * v.y + this.m23 * v.z + this.m24 * v.w,
691                            this.m31 * v.x + this.m32 * v.y + this.m33 * v.z + this.m34 * v.w,
692                            this.m41 * v.x + this.m42 * v.y + this.m43 * v.z + this.m44 * v.w);
693     },
694     
695     /**
696     * Multiples the matrix by the column vector v. It is assumed the Vector3 v value
697     * is equivalent to a Vector4 instance with a w value of 0
698     * @param {Vector3} v input vector
699     * @return {Vector3}
700     */
701     transformVector3: function (v) {
702         return new Vector3(this.m11 * v.x + this.m12 * v.y + this.m13 * v.z,
703                            this.m21 * v.x + this.m22 * v.y + this.m23 * v.z,
704                            this.m31 * v.x + this.m32 * v.y + this.m33 * v.z);
705     },
706     
707     /**
708     * Returns the determinant of the calling matrix
709     * @return {number}
710     */
711     determinant: function () {
712         var a, b, c, d, e, f, g, h, i, j, k, l;
713         a = this.m11 * this.m22 - this.m12 * this.m21;
714         b = this.m11 * this.m23 - this.m13 * this.m21;
715         c = this.m11 * this.m24 - this.m14 * this.m21;
716         d = this.m12 * this.m23 - this.m13 * this.m22;
717         e = this.m12 * this.m24 - this.m14 * this.m22;
718         f = this.m13 * this.m24 - this.m14 * this.m23;
719         g = this.m31 * this.m42 - this.m32 * this.m41;
720         h = this.m31 * this.m43 - this.m33 * this.m41;
721         i = this.m31 * this.m44 - this.m34 * this.m41;
722         j = this.m32 * this.m43 - this.m33 * this.m42;
723         k = this.m32 * this.m44 - this.m34 * this.m42;
724         l = this.m33 * this.m44 - this.m34 * this.m43;
725         return a * l - b * k + c * j + d * i - e * h + f * g;
726     },
727     
728     /**
729     * Returns the inverse of the calling matrix.  If the matrix cannot be inverted
730     * the the identity matrix is returned.
731     * @return {Matrix4x4}
732     */
733     inverse: function () {
734         var a, b, c, d, e, f, g, h, i, j, k, l, determinant, invD, 
735             m11, m12, m13, m14, m21, m22, m23, m24, m31, m32, m33, m34, m41, m42, m43, m44;
736             
737         a = this.m11 * this.m22 - this.m12 * this.m21;
738         b = this.m11 * this.m23 - this.m13 * this.m21;
739         c = this.m11 * this.m24 - this.m14 * this.m21;
740         d = this.m12 * this.m23 - this.m13 * this.m22;
741         e = this.m12 * this.m24 - this.m14 * this.m22;
742         f = this.m13 * this.m24 - this.m14 * this.m23;
743         g = this.m31 * this.m42 - this.m32 * this.m41;
744         h = this.m31 * this.m43 - this.m33 * this.m41;
745         i = this.m31 * this.m44 - this.m34 * this.m41;
746         j = this.m32 * this.m43 - this.m33 * this.m42;
747         k = this.m32 * this.m44 - this.m34 * this.m42;
748         l = this.m33 * this.m44 - this.m34 * this.m43;
749         determinant =  a * l - b * k + c * j + d * i - e * h + f * g;
750         if (MathHelper.abs(determinant) < MathHelper.zeroTolerance)
751         {
752             return Matrix4x4.createIdentity();
753         }
754         
755         m11 = this.m22 * l - this.m23 * k + this.m24 * j;
756         m12 = -this.m12 * l + this.m13 * k - this.m14 * j;
757         m13 = this.m42 * f - this.m43 * e + this.m44 * d;
758         m14 = -this.m32 * f + this.m33 * e - this.m34 * d;
759         
760         m21 = -this.m21 * l + this.m23 * i - this.m24 * h;
761         m22 = this.m11 * l - this.m13 * i + this.m14 * h;
762         m23 = -this.m41 * f + this.m43 * c - this.m44 * b;
763         m24 = this.m31 * f - this.m33 * c + this.m34 * b;
764         
765         m31 = this.m21 * k - this.m22 * i + this.m24 * g;
766         m32 = -this.m11 * k + this.m12 * i - this.m14 * g;
767         m33 = this.m41 * e - this.m42 * c + this.m44 * a;
768         m34 = -this.m31 * e + this.m32 * c - this.m34 * a;
769         
770         m41 = -this.m21 * j + this.m22 * h - this.m23 * g;
771         m42 = this.m11 * j - this.m12 * h + this.m13 * g;
772         m43 = -this.m41 * d + this.m42 * b - this.m43 * a;
773         m44 = this.m31 * d - this.m32 * b + this.m33 * a;
774         invD = 1.0 / determinant;
775         return new Matrix4x4(m11 * invD, m12 * invD, m13 * invD, m14 * invD,
776                              m21 * invD, m22 * invD, m23 * invD, m24 * invD,
777                              m31 * invD, m32 * invD, m33 * invD, m34 * invD,
778                              m41 * invD, m42 * invD, m43 * invD, m44 * invD);
779     },
780 
781     /**
782     * Returns a string containing the current state of the matrix.  Useful for debugging purposes
783     * @return {string}
784     */
785     toString: function () {
786         return this.m11 + ", " + this.m12 + ", " + this.m13 + ", " + this.m14 + "\n" +
787                this.m21 + ", " + this.m22 + ", " + this.m23 + ", " + this.m24 + "\n" +
788                this.m31 + ", " + this.m32 + ", " + this.m33 + ", " + this.m34 + "\n" +
789                this.m41 + ", " + this.m42 + ", " + this.m43 + ", " + this.m44 + "\n";
790     }
791 };

807 /**
808 * Creates a Quaternion from a rotation matrix
809 * @param {Matrix4x4} m rotation matrix  
810 * @return {Quaternion} 
811 */
812 Quaternion.fromRotationMatrix = function (m) {
813     //See: http://www.euclideanspace.com/maths/geometry/rotations/conversions/matrixToQuaternion/index.htm
814 
815 	/*jslint onevar: false */
816 	//JSLint and I don't agree, as type annotions mean we need multiple vars.
817 
818 	/** @type {number} */
819     var trace; 
820 
821 	/** @type {number} */
822 	var temp; 
823 
824 	/** @type {Quaternion} */
825 	var result;
826 
827 	/** @type {number} */
828 	var largestIndex;
829     
830     
831     result = new Quaternion(0, 0, 0, 0);
832     trace = m.m11 + m.m22 + m.m33;
833     if (MathHelper.abs(trace) > MathHelper.zeroTolerance)
834     {
835         result.w = MathHelper.sqrt(trace + 1) * 0.5;
836         temp = 1.0 / (4 * result.w);
837         result.x = (m.m32 - m.m23) * temp;
838         result.y = (m.m13 - m.m31) * temp;
839         result.z = (m.m21 - m.m12) * temp;
840     }
841     else
842     {
843         largestIndex = 0;
844         if (m.m22 > m.m11)
845         {
846             largestIndex = 1;
847             if (m.m33 > m.m22)
848             {
849                 largestIndex = 2;
850             }
851         }
852         else if (m.m33 > m.m11)
853         {
854             largestIndex = 2;
855         }
856        
857 	    switch (largestIndex)
858 		{
859 		case 0:
860 			result.x = 0.5 * MathHelper.sqrt(m.m11 - m.m22 - m.m33 + 1);
861 			temp = 1.0 / (4 * result.x);
862 			result.w = (m.m32 - m.m23) * temp;
863 			result.y = (m.m12 + m.m21) * temp;
864 			result.z = (m.m13 + m.m31) * temp;
865 			break;
866 		case 1:
867 			result.y = 0.5 * MathHelper.sqrt(m.m22 - m.m11 - m.m33 + 1);
868 			temp = 1.0 / (4 * result.y);
869 			result.w = (m.m13 - m.m31) * temp;
870 			result.x = (m.m12 + m.m21) * temp;
871 			result.z = (m.m23 + m.m32) * temp;
872 			break;
873 		case 2:
874 			result.z = 0.5 * MathHelper.sqrt(m.m33 - m.m11 - m.m22 + 1);
875 			temp = 1.0 / (4 * result.z);
876 			result.w = (m.m21 - m.m12) * temp;
877 			result.x = (m.m13 + m.m31) * temp;
878 			result.y = (m.m32 + m.m23) * temp;
879 			break;
880 		}
881     }
882     return result;
883 };

885 /**
886 * Creates a Quaternion from an axis and an angle
887 * @param {Vector3} axis The rotation axis, must be a unit vector
888 * @param {number} angle An angle in radians.  A positive angle will rotate anticlockwise around the axis
889 * @return {Quaternion}
890 */
891 Quaternion.fromAxisAngle = function (axis, angle) {
892     var halfAngle, s;
893     halfAngle = 0.5 * angle;
894     s = MathHelper.sin(halfAngle);
895     return new Quaternion(MathHelper.cos(halfAngle), axis.x * s, axis.y * s, axis.z * s);
896 };

898 /**
899 * Returns a quaternion that has been slerped between source and target by t amount
900 * @param {number} t A value between 0.0 and 1.0 inclusive
901 * @param {Quaternion} source The starting quaternion value
902 * @param {Quaternion} target The target quaternion value
903 * @return {Quaternion}
904 */
905 Quaternion.slerp = function (t, source, target) {
906     var cos, angle, sin, invSin, a, b;
907     
908     if (t === 0.0)
909     {
910         return source;
911     }
912     if (t === 1.0)
913     {
914         return target;
915     }
916     
917     cos = source.dot(target);
918     angle = MathHelper.acos(cos);
919     if (MathHelper.abs(angle) >= MathHelper.zeroTolerance)
920     {
921         sin = MathHelper.sin(angle);
922         invSin = 1.0 / sin;
923         a = MathHelper.sin((1.0 - t) * angle) * invSin;
924         b = MathHelper.sin(t * angle) * invSin;
925         return source.multiplyScalar(a).add(target.multiplyScalar(b));
926     }
927     
928     return source;
929 };

931 Quaternion.prototype = 
932 {
933     /**
934     * Returns the dot product of two Quaternions
935     * @param {Quaternion} q input quaternion
936     * @return {number}
937     */
938     dot: function (q) {
939         return this.w * q.w + this.x * q.x + this.y * q.y + this.z * q.z;
940     },
941     
942     /**
943     * Calculates the length of the Quaternion
944     * @return {number}
945     */
946     length: function () {
947         return MathHelper.sqrt(this.w * this.w + this.x * this.x + this.y * this.y + this.z * this.z);
948     },
949     
950     /**
951     * Creates a unit length version of the Quaternion
952     * @return {Quaternion}
953     */
954     normalize: function () {
955         var length, inverseLength;
956         
957         length = this.length();
958         if (length < MathHelper.zeroTolerance) {
959             return new Quaternion(0.0, 0.0, 0.0, 0.0);
960         }
961 
962         inverseLength = 1.0 / length;
963         return new Quaternion(this.w * inverseLength,
964                               this.x * inverseLength,
965                               this.y * inverseLength,
966                               this.z * inverseLength);
967     },
968     
969     /**
970     * Returns the inverse of the calling quaternion
971     * @return {Quaternion} If the Quaternion cannot be inversed, a Quaternion with x,y,z,w == 0.0 is returned
972     */
973     inverse: function () {
974         var norm, invNorm;
975         
976         norm = this.w * this.w + this.x * this.x + this.y * this.y * this.z * this.z;
977         if (MathHelper.abs(norm) > MathHelper.zeroTolerance)
978         {
979             invNorm = 1.0 / norm;
980             return new Quaternion(this.w * invNorm,
981                                   -this.x * invNorm,
982                                   -this.y * invNorm,
983                                   -this.z * invNorm);
984         }
985         return new Quaternion(0.0, 0.0, 0.0, 0.0);
986     },
987     
988     /**
989     * Returns the conjugate of the quaternion
990     * @return {Quaternion}
991     */
992     conjugate: function () {
993         return new Quaternion(this.w, -this.x, -this.y, -this.z);
994     },
995     
996     /**
997     * Applies the quaternion rotation to the input vector and returns the result
998     * @param {Vector3} v input vector to be rotated
999     * @return {Vector3} The rotated vector
1000     */
1001     transform: function (v) {
1002         
1003         //See Bishop, Van Verth (make sure to look at the errata)
1004         var p, d, c;
1005         d = 2.0 * (this.x * v.x + this.y * v.y + this.z * v.z);
1006         c = 2.0 * this.w;
1007         p = c * this.w - 1.0;
1008         return new Vector3(p * v.x + d * this.x + c * (this.y * v.z - this.z * v.y),
1009                            p * v.y + d * this.y + c * (this.z * v.x - this.x * v.z),
1010                            p * v.z + d * this.z + c * (this.x * v.y - this.y * v.x));
1011     },
1012     
1013     add: function (q) {
1014         return new Quaternion(this.w + q.w, this.x + q.x, this.y + q.y, this.z + q.z);
1015     },
1016     
1017     /**
1018     * Returns a Quaternion representing the result of multiplying the calling Quaternion by Quaternion q
1019     * like q2 * q1 where q1 is applied before q2
1020     * @param {Quaternion} q input quaternion
1021     * @return {Quaternion}
1022     */
1023     multiply: function (q) {
1024         return new Quaternion(this.w * q.w - this.x * q.x - this.y * q.y - this.z * q.z,
1025                               this.y * q.z - this.z * q.y + this.w * q.x + q.w * this.x,
1026                               this.z * q.x - this.x * q.z + this.w * q.y + q.w * this.y,
1027                               this.x * q.y - this.y * q.x + this.w * q.z + q.w * this.z);
1028     },
1029     
1030     /**
1031     * Multiplies each element in the Quaternion by the scalar f
1032     * @param {number} f input scalar
1033     * @return {Quaternion}
1034     */
1035     multiplyScalar: function (f) {
1036         return new Quaternion(this.w * f, this.x * f, this.y * f, this.z * f);
1037     },
1038     
1039     /**
1040     * Converts the quaternion to a rotation matrix
1041     * @return {Matrix4x4}
1042     */
1043     toRotationMatrix: function () {
1044         var x, y, z, wx, wy, wz, xx, xy, xz, yy, yz, zz;
1045         x = 2.0 * this.x;
1046         y = 2.0 * this.y;
1047         z = 2.0 * this.z;
1048         wx = x * this.w;
1049         wy = y * this.w;
1050         wz = z * this.w;
1051         xx = x * this.x;
1052         xy = y * this.x;
1053         xz = z * this.x;
1054         yy = y * this.y;
1055         yz = z * this.y;
1056         zz = z * this.z;
1057         
1058         //TODO: When Matrix3x3 exists change this to a 3x3 instead
1059         return new Matrix4x4(1.0 - (yy + zz), xy - wz, xz + wy, 0,
1060                              xy + wz, 1.0 - (xx + zz), yz - wx, 0,
1061                              xz - wy, yz + wx, 1.0 - (xx + yy), 0,
1062                              0, 0, 0, 1);
1063     },
1064     
1065     /**
1066     * Converts the Quaternion to an axis and an angle
1067     * @return {Vector4} Containing the values x,y,z,angle where angle is in radians
1068     */
1069     toAxisAngle: function () {
1070         var lengthSquared, inverseLength;
1071         
1072         lengthSquared = this.x * this.x + this.y * this.y + this.z * this.z;
1073         if (lengthSquared > MathHelper.zeroTolerance)
1074         {
1075             inverseLength = MathHelper.invSqrt(lengthSquared);
1076             return new Vector4(this.x * inverseLength, this.y * inverseLength, this.z * inverseLength, 2.0 * MathHelper.acos(this.w));
1077         }
1078         return new Vector4(1, 0, 0, 0);
1079     },
1080     
1081     /**
1082     * Returns a string containing the current state of the Quaternion
1083     * @return {string}
1084     */
1085     toString: function () {
1086         return '[' + this.w + ', ' + this.x + ', ' + this.y + ', ' + this.z + ']';
1087     }
